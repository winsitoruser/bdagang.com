================================================================================
PROPOSED SHIPPING WORKFLOW
================================================================================

COMPLETE FLOW:

1. REQUISITION APPROVED
   Status: approved
   Action: HQ assigns fulfilling branch
   Next: Fulfilling branch prepares items

2. PREPARATION (Processing)
   Status: processing
   Actions:
   - Fulfilling branch picks items
   - Verifies quantities
   - Packs items
   - Generates packing list
   - Creates shipping label
   Next: Mark as ready to ship

3. READY TO SHIP
   Status: ready_to_ship
   Actions:
   - Final verification
   - Assign carrier/driver
   - Generate tracking number
   - Create delivery note
   - Schedule pickup/delivery
   Next: Ship items

4. IN TRANSIT
   Status: in_transit
   Actions:
   - Track shipment location
   - Update ETA
   - Monitor delays
   - Send notifications
   Next: Deliver to destination

5. DELIVERED
   Status: delivered
   Actions:
   - Confirm receipt
   - Verify quantities
   - Check condition
   - Capture signature/photo
   - Report discrepancies
   Next: Complete requisition

6. COMPLETED
   Status: completed
   Actions:
   - Close requisition
   - Update inventory
   - Generate reports
   - Archive documents

DATABASE SCHEMA DESIGN
================================================================================

NEW TABLE: requisition_shipments
CREATE TABLE requisition_shipments (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  requisition_id UUID NOT NULL REFERENCES internal_requisitions(id),
  shipment_number VARCHAR(50) UNIQUE NOT NULL,
  tracking_number VARCHAR(100),
  carrier VARCHAR(100), -- 'internal', 'jne', 'jnt', 'sicepat', etc.
  driver_name VARCHAR(255),
  driver_phone VARCHAR(50),
  vehicle_number VARCHAR(50),
  
  -- Shipping details
  shipped_from_branch_id UUID REFERENCES branches(id),
  shipped_to_branch_id UUID REFERENCES branches(id),
  shipped_by INTEGER REFERENCES users(id),
  shipped_at TIMESTAMP,
  
  -- Delivery details
  estimated_delivery_date DATE,
  actual_delivery_date TIMESTAMP,
  delivery_instructions TEXT,
  
  -- Package details
  total_packages INTEGER DEFAULT 1,
  total_weight DECIMAL(10,2), -- in kg
  total_volume DECIMAL(10,2), -- in m3
  
  -- Costs
  shipping_cost DECIMAL(15,2) DEFAULT 0,
  insurance_cost DECIMAL(15,2) DEFAULT 0,
  
  -- Status
  status VARCHAR(50) DEFAULT 'pending',
  -- pending, picked_up, in_transit, out_for_delivery, delivered, failed, returned
  
  -- Documents
  packing_list_url TEXT,
  delivery_note_url TEXT,
  shipping_label_url TEXT,
  
  notes TEXT,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_shipments_requisition ON requisition_shipments(requisition_id);
CREATE INDEX idx_shipments_tracking ON requisition_shipments(tracking_number);
CREATE INDEX idx_shipments_status ON requisition_shipments(status);

---

NEW TABLE: shipment_tracking_updates
CREATE TABLE shipment_tracking_updates (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  shipment_id UUID NOT NULL REFERENCES requisition_shipments(id),
  status VARCHAR(50) NOT NULL,
  location VARCHAR(255),
  latitude DECIMAL(10,8),
  longitude DECIMAL(11,8),
  description TEXT,
  updated_by INTEGER REFERENCES users(id),
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_tracking_shipment ON shipment_tracking_updates(shipment_id);
CREATE INDEX idx_tracking_created ON shipment_tracking_updates(created_at);

---

NEW TABLE: delivery_confirmations
CREATE TABLE delivery_confirmations (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  shipment_id UUID NOT NULL REFERENCES requisition_shipments(id),
  requisition_id UUID NOT NULL REFERENCES internal_requisitions(id),
  
  -- Receiver info
  received_by_name VARCHAR(255) NOT NULL,
  received_by_user_id INTEGER REFERENCES users(id),
  received_at TIMESTAMP NOT NULL DEFAULT NOW(),
  
  -- Verification
  items_verified BOOLEAN DEFAULT false,
  condition_good BOOLEAN DEFAULT true,
  all_items_received BOOLEAN DEFAULT true,
  
  -- Discrepancies
  has_discrepancies BOOLEAN DEFAULT false,
  missing_items JSONB, -- [{productId, name, quantityMissing}]
  damaged_items JSONB, -- [{productId, name, quantityDamaged, description}]
  extra_items JSONB, -- [{productId, name, quantityExtra}]
  
  -- Proof
  signature_url TEXT,
  photo_urls JSONB, -- Array of photo URLs
  
  notes TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_delivery_shipment ON delivery_confirmations(shipment_id);
CREATE INDEX idx_delivery_requisition ON delivery_confirmations(requisition_id);

---

NEW TABLE: shipment_items
CREATE TABLE shipment_items (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  shipment_id UUID NOT NULL REFERENCES requisition_shipments(id),
  requisition_item_id UUID REFERENCES internal_requisition_items(id),
  product_id INTEGER NOT NULL,
  product_name VARCHAR(255) NOT NULL,
  sku VARCHAR(100),
  
  quantity_shipped DECIMAL(15,2) NOT NULL,
  quantity_received DECIMAL(15,2),
  unit VARCHAR(50),
  
  package_number INTEGER, -- Which package this item is in
  
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_shipment_items_shipment ON shipment_items(shipment_id);

ALTER TABLE internal_requisitions

API ENDPOINTS DESIGN
================================================================================

SHIPPING MANAGEMENT:

1. POST /api/hq/requisitions/:id/create-shipment
   Purpose: Create shipment for approved requisition
   Body: {
     carrier, driverName, driverPhone, vehicleNumber,
     estimatedDeliveryDate, totalPackages, shippingCost,
     deliveryInstructions, items: [{productId, quantity, packageNumber}]
   }
   Response: Created shipment with shipment_number

2. PUT /api/hq/requisitions/:id/shipments/:shipmentId/mark-ready
   Purpose: Mark shipment as ready to ship
   Body: { packingListUrl, deliveryNoteUrl, shippingLabelUrl }
   Response: Updated shipment

3. PUT /api/hq/requisitions/:id/shipments/:shipmentId/ship
   Purpose: Mark shipment as shipped (in transit)
   Body: { trackingNumber, shippedAt, shippedBy }
   Response: Updated shipment with tracking

4. GET /api/shipments/:shipmentId/tracking
   Purpose: Get shipment tracking history
   Response: Array of tracking updates

5. POST /api/shipments/:shipmentId/tracking
   Purpose: Add tracking update
   Body: { status, location, latitude, longitude, description }
   Response: Created tracking update

6. POST /api/shipments/:shipmentId/confirm-delivery
   Purpose: Confirm delivery and receive items
   Body: {
     receivedByName, receivedByUserId,
     itemsVerified, conditionGood, allItemsReceived,
     missingItems, damagedItems, extraItems,
     signatureUrl, photoUrls, notes
   }
   Response: Created delivery confirmation

7. GET /api/hq/shipments/active
   Purpose: Get all active shipments (monitoring)
   Query: ?status=in_transit&branch=uuid
   Response: Array of active shipments with tracking

8. GET /api/hq/shipments/analytics
   Purpose: Get shipping analytics
   Query: ?dateFrom=2026-01-01&dateTo=2026-12-31
   Response: Shipping metrics and KPIs

BRANCH ENDPOINTS:

9. GET /api/branch/requisitions/:id/shipment
   Purpose: Get shipment details for branch requisition
   Response: Shipment with tracking updates

10. POST /api/branch/shipments/:shipmentId/receive
    Purpose: Branch receives and confirms delivery
    Body: Same as confirm-delivery
    Response: Delivery confirmation

================================================================================
WORKFLOW IMPLEMENTATION
================================================================================

PROCESS 1: CREATE SHIPMENT (HQ)

1. HQ user views approved requisition
2. Clicks "Create Shipment" button
3. Modal opens with shipment form
4. Selects carrier type (internal/external)
5. Fills driver details (if internal)
6. Sets delivery date and package info
7. Reviews items to ship
8. Assigns items to packages
9. Adds delivery instructions
10. Clicks "Create Shipment"
11. API creates shipment record
12. Generates shipment number (SH-XXXXX)
13. Status: pending
14. Notification sent to fulfilling branch

PROCESS 2: PREPARE SHIPMENT (Fulfilling Branch)

1. Branch receives notification
2. Views shipment details
3. Picks items from inventory
4. Packs items into packages
5. Generates packing list (PDF)
6. Creates delivery note
7. Prints shipping label
8. Uploads documents
9. Clicks "Mark Ready to Ship"
10. Status: ready_to_ship
11. Notification to HQ and driver

PROCESS 3: SHIP ITEMS (Fulfilling Branch/Driver)

1. Driver picks up packages
2. Branch user clicks "Ship Now"
3. Enters tracking number (if external carrier)
4. Confirms shipped time
5. Status: in_transit
6. Initial tracking update created
7. Notification to requesting branch
8. ETA calculated and displayed

PROCESS 4: TRACK SHIPMENT (All Users)

1. User views shipment tracking
2. Sees real-time location updates
3. Views tracking history timeline
4. Checks ETA
5. Receives notifications for:
   - Picked up
   - In transit
   - Delayed
   - Out for delivery
   - Delivered

PROCESS 5: RECEIVE & CONFIRM (Requesting Branch)

1. Driver arrives at branch
2. Branch user opens delivery confirmation
3. Verifies items against packing list
4. Counts quantities
5. Checks condition
6. Reports any discrepancies:
   - Missing items
   - Damaged items
   - Extra items
7. Captures signature
8. Takes photos (optional)
9. Adds notes
10. Clicks "Confirm Delivery"
11. Status: delivered
12. Delivery confirmation created
13. Inventory updated
14. Notification to HQ and fulfilling branch

PROCESS 6: COMPLETE REQUISITION (System)

1. All shipments delivered
2. All items confirmed
3. Discrepancies resolved (if any)
4. Status: completed
5. Generate completion report
6. Archive documents
7. Update analytics
